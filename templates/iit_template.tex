% Section heading with full-width rule.
% Uses \leaders\hrule so the rule never fights with \vspace.
\newcommand{\resheading}[1]{%
  \vspace{7pt}%
  \noindent\textbf{\large #1}\,%
  \leaders\hrule height 0.55pt\hfill\kern0pt%
  \par\vspace{4pt}%
}

% Sub-heading row: organisation + date on line 1, role/degree + location/score on line 2.
% Positive \vspace{3pt} below keeps bullets from gluing to the heading.
\newcommand{\ressubheading}[4]{%
  \begin{tabular*}{\textwidth}{@{}l@{\extracolsep{\fill}}r@{}}
    \textbf{#1} & \small #2 \\[-1pt]
    \textit{\small #3} & \textit{\small #4} \\
  \end{tabular*}%
  \vspace{3pt}%
}

% Bullet item — small font, 1pt padding below so bullets don't crowd.
\newcommand{\resitem}[1]{%
  \item \small #1 \vspace{1pt}%
}
